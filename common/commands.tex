\newcommand{\ruText}[1]{
    {\footnotesize \textcolor{darkgray}{#1} \par}
}

    \textbf{Кантрольная работа па тэме <<#1>> (варыянт #3)} \\
    \ruText{Контрольная работа по теме <<#2>> (вариант #3)}
\end{center}
}

\newcommand{\problemItemSimple}[3][]{%
    \ifthenelse{\equal{#1}{*}}{\addtocounter{enumi}{1}}{}%
    \ifthenelse{\equal{#1}{*}}{\item[\theenumi*.] #2 \\}{\item #2 \\}%
    \ruText{#3}%
}

\newcommand{\problemItemWithCommonPart}[4][]{%
    \ifthenelse{\equal{#1}{*}}{\addtocounter{enumi}{1}}{}%
    \ifthenelse{\equal{#1}{*}}{\item[\theenumi*.] #2 \\}{\item #2 \\}%
    \ruText{#3}%
    #4%
}

\newcommand{\problemItemWithCommonPartComplicated}[6][]{%
    \ifthenelse{\equal{#1}{*}}{\addtocounter{enumi}{1}}{}%
    \ifthenelse{\equal{#1}{*}}{\item[\theenumi*.] #2 \\}{\item #2 \\}%
    \ruText{#3}%
    #4 \textbf{}\\
    #5 \\%
    \ruText{#6}%
}
